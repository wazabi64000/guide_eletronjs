4. Multi-fenêtres

const windows = new Map();

function createEditorWindow(filePath) {
  const win = new BrowserWindow({ /* ... */ });
  windows.set(win.id, win);
  win.on('closed', () => windows.delete(win.id));
  win.loadFile('editor.html', { query: { file: filePath } });
}
